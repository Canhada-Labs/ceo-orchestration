#: Verdict values per ADR-106 / ADR-108 (Pair-Rail asymmetric VETO).
#: This is the trust-boundary allowlist; it MUST remain in the kernel.
_VALID_VERDICTS: Tuple[str, ...] = ("PASS", "ADVISORY", "BLOCK")

#: Strict-JSON / last-message size cap (256 KB). Trust-boundary DoS guard;
#: STAYS in the kernel. (Was PLAN-086 Wave A.6 Sec-1 fold.)
_STRICT_JSON_SIZE_CAP_BYTES = 256 * 1024

# ENVELOPE STRIPPING + FINDING NORMALIZATION — STAY IN THE KERNEL.
# (Reproduced unchanged from the current file so the parsers below resolve;
#  execution session: these already exist, do NOT duplicate — keep one copy.)

_ANSI_ESCAPE_RE = re.compile(r"\x1b\[[0-9;]*[A-Za-z]")
_MD_FENCE_RE = re.compile(r"^```(?:json)?\s*\n", re.MULTILINE)
_MD_FENCE_END_RE = re.compile(r"\n```\s*$", re.MULTILINE)


def _strip_codex_envelope(stdout: str) -> str:
    """Strip ANSI / markdown fence wrappers around Codex JSON output.

    Retained verbatim. The 0.139 ``-o`` last-message file is normally a bare
    JSON object (verified: a one-line JSON object), but the agent prompt or a
    chatty model may still wrap it in a ```json fence; stripping stays cheap
    and defensive.
    """
    s = _ANSI_ESCAPE_RE.sub("", stdout)
    s = _MD_FENCE_RE.sub("", s)
    s = _MD_FENCE_END_RE.sub("", s)
    return s.strip()


def _normalize_finding(item: Dict[str, Any]) -> Dict[str, Any]:
    """Coerce a raw finding dict to the normalized shape. (Unchanged.)"""
    severity = item.get("severity")
    if severity not in ("P0", "P1"):
        severity = "P1"  # default conservative
    return {
        "rubric_violation_id": str(item.get("rubric_violation_id") or "RV-UNKNOWN"),
        "severity": severity,
        "file": str(item.get("file") or ""),
        "line": int(item.get("line") or 0) if str(item.get("line") or "").isdigit() else 0,
        "rationale": str(item.get("rationale") or ""),
    }


# ===========================================================================
# NEW SHARED HELPER — read ONE structured verdict object from the -o
# last-message FILE CONTENT (already redacted by the caller).
# ===========================================================================
#
# PLAN-142 D1: the 0.139 ``-o/--output-last-message <FILE>`` writes ONLY the
# final agent message (verified hands-on: a one-line JSON object). The verdict
# therefore lives in ONE structured object — NOT in the JSONL ``--json`` event
# stream, and NOT in whole-stdout. This helper does the single ``json.loads``
# of that (post-redaction) content and the schema-shape validation that BOTH
# parse_verdict and parse_verdict_strict share. It is the single chokepoint
# where the untrusted object becomes a trusted, normalized dict.
#
# IMPORTANT trust-boundary contract:
#   - INPUT is the ALREADY-REDACTED last-message content (a str). Redaction is
#     the CALLER's job (check_pair_rail.py redacts the full byte-string before
#     calling; the in-process promotion path redacts in codex_invoke.py). This
#     helper does NOT itself redact — keeping redaction at the single egress
#     callsite per ADR-114 (no double-redaction, no second pass). See open_items
#     for the audit that every caller redacts first.
#   - It enforces the 256 KB cap (DoS guard) and rejects non-dict / bad-verdict.
#   - It returns a normalized dict on success, or a typed failure marker the
#     two public parsers translate into their respective contracts (ADR-106
#     fail-OPEN for parse_verdict; fail-CLOSED-to-ADVISORY for
#     parse_verdict_strict — see DESIGN NOTE A).


def _read_structured_verdict_object(content: Optional[str]) -> Dict[str, Any]:
    """Parse a single structured verdict object from -o last-message content.

    ``content`` is the (already egress-redacted) text of the codex-cli 0.139
    ``-o/--output-last-message`` file — exactly ONE JSON object, the final
    agent message. NOT the ``--json`` JSONL event stream, NOT whole stdout.

    Returns a dict ALWAYS containing:
      - ``ok``       (bool)        — True iff a valid, schema-shaped verdict.
      - ``verdict``  (Optional[str]) — the raw verdict string IF present
                                       (may be invalid; caller decides policy).
      - ``findings`` (List[Dict])  — normalized findings (empty on miss).
      - ``summary``  (str)
      - ``error``    (Optional[str]) — short, payload-free reason on failure.

    Trust rules enforced here (single chokepoint):
      - 256 KB byte cap on the input (DoS / runaway-output guard).
      - top-level MUST be a JSON object (dict).
      - ``verdict`` MUST be in ``_VALID_VERDICTS`` for ``ok=True``.
      - ``findings`` (if present) MUST be a list of dicts; each normalized.
      - ``summary`` (if present) MUST be a str; else coerced to "".

    NEVER raises. NEVER returns or logs the raw object contents in ``error``
    (payload-free), so a malformed-but-secret-bearing message cannot leak via
    the error string.
    """
    # --- presence / type guard ------------------------------------------
    if not content or not isinstance(content, str):
        return {
            "ok": False,
            "verdict": None,
            "findings": [],
            "summary": "",
            "error": "[codex] empty last-message content",
        }

    # --- size cap (trust-boundary DoS guard) ----------------------------
    # Use errors="replace" so an undecodable tail cannot raise here.
    if len(content.encode("utf-8", errors="replace")) > _STRICT_JSON_SIZE_CAP_BYTES:
        return {
            "ok": False,
            "verdict": None,
            "findings": [],
            "summary": "",
            "error": "[codex] last-message exceeds 256 KB cap",
        }

    cleaned = _strip_codex_envelope(content)
    if not cleaned:
        return {
            "ok": False,
            "verdict": None,
            "findings": [],
            "summary": "",
            "error": "[codex] last-message empty after envelope strip",
        }

    # --- the SINGLE json.loads of ONE object (not JSONL) ----------------
    try:
        obj = json.loads(cleaned)
    except (json.JSONDecodeError, ValueError) as e:
        # Payload-free error: include only decoder metadata, never the body.
        lineno = getattr(e, "lineno", "?")
        return {
            "ok": False,
            "verdict": None,
            "findings": [],
            "summary": "",
            "error": "[codex] last-message JSON decode at line {0}".format(lineno),
        }

    if not isinstance(obj, dict):
        return {
            "ok": False,
            "verdict": None,
            "findings": [],
            "summary": "",
            "error": "[codex] last-message top-level not a JSON object",
        }

    # --- normalize findings + summary regardless of verdict validity ----
    findings_raw = obj.get("findings", [])
    findings: List[Dict[str, Any]] = []
    if isinstance(findings_raw, list):
        for fitem in findings_raw:
            if isinstance(fitem, dict):
                findings.append(_normalize_finding(fitem))
            # Non-dict finding items are silently dropped (fail-permissive on
            # the findings list shape; the verdict is the load-bearing field).

    summary = obj.get("summary", "")
    if not isinstance(summary, str):
        summary = ""

    verdict = obj.get("verdict")
    if verdict not in _VALID_VERDICTS:
        # Object parsed but verdict is absent/invalid. ok=False; the caller
        # applies its contract (ADR-106 coerce vs fail-closed). We surface the
        # presence of an object so callers can distinguish "no object" from
        # "object without a valid verdict" if they ever need to.
        return {
            "ok": False,
            "verdict": verdict if isinstance(verdict, str) else None,
            "findings": findings,
            "summary": summary,
            "error": "[codex] verdict not in _VALID_VERDICTS",
        }

    return {
        "ok": True,
        "verdict": verdict,
        "findings": findings,
        "summary": summary,
        "error": None,
    }


# ===========================================================================
# parse_verdict_strict — LIVE-RAIL consume path (check_pair_rail.py).
# ===========================================================================
#
# DESIGN NOTE A (PLAN-142 §3 fail-CLOSED-to-ADVISORY):
# The plan's live-rail AC says: read the -o last-message file -> single-pass
# redact the FULL byte-string -> parse_verdict_strict (verdict in
# _VALID_VERDICTS, fail-CLOSED-to-ADVISORY) -> decide().
#
# "fail-CLOSED-to-ADVISORY" means: on ANY parse/shape failure the rail does NOT
# treat the response as a passing/blocking verdict and does NOT crash — it
# degrades to ADVISORY (the conservative neutral that never silently passes a
# write and never hard-blocks on infra). Therefore the NEW strict parser
# RETURNS a normalized ADVISORY dict on failure rather than raising typed
# exceptions. This is a DELIBERATE behavior change from the current
# exception-raising parse_verdict_strict (which was built for the old
# --strict-json whole-stdout path that the live rail never actually called).
#
#   * Rationale for not raising: the live rail must never raise into decide();
#     a degradation matrix (file missing / empty / oversize / present-after-
#     exit!=0 / symlink-refused) all map to ADVISORY at the call site. Having
#     the parser ALSO raise would force the rail to wrap every call in
#     try/except and re-map to ADVISORY anyway — the parser owning the
#     fail-closed-to-ADVISORY contract is cleaner and matches parse_verdict's
#     never-raise shape.
#   * The old typed exceptions (CodexResponseTooLarge / CodexJsonInvalid) are
#     NO LONGER RAISED by this function. They remain defined in
#     _lib/exceptions.py (other callers/tests may still reference the types).
#     See open_items: confirm no remaining caller `except`s them from this
#     function. The import of those types is DROPPED from this function body.
#
# Signature note: the parameter is renamed `last_message` (was `stdout`) to
# make the new contract explicit — it is the -o FILE CONTENT, already redacted,
# NOT raw stdout. This is a kernel-internal rename; the sole live caller is
# check_pair_rail.py (also edited this ceremony) — keep them in sync.


def parse_verdict_strict(last_message: str) -> Dict[str, Any]:
    """Strict verdict parse for the LIVE pair-rail consume path (0.139 `-o`).

    PLAN-142 §3 rewrite. ``last_message`` is the ALREADY-REDACTED content of
    the codex-cli 0.139 ``-o/--output-last-message`` file — exactly one JSON
    object (the final agent message). This function:

      1. Reads the single structured object via ``_read_structured_verdict_object``.
      2. Requires ``verdict in _VALID_VERDICTS`` for a "real" verdict.
      3. fail-CLOSED-to-ADVISORY: on ANY miss (empty / oversize / non-JSON /
         non-object / bad verdict) returns a normalized ADVISORY envelope with
         ``parse_error`` set — it NEVER raises and NEVER returns PASS/BLOCK on
         a malformed response (forged free-text 'PASS' with no structured
         verdict -> ADVISORY, satisfying R-SEC-2).

    Returns a dict ALWAYS containing:
      ``verdict`` (str in _VALID_VERDICTS), ``findings`` (List[Dict]),
      ``summary`` (str), ``parse_error`` (Optional[str]).

    NEVER raises.
    """
    result = _read_structured_verdict_object(last_message)
    if result["ok"]:
        return {
            "verdict": result["verdict"],
            "findings": result["findings"],
            "summary": result["summary"],
            "parse_error": None,
        }
    # fail-CLOSED-to-ADVISORY: any miss -> ADVISORY, never PASS/BLOCK, no raise.
    return {
        "verdict": "ADVISORY",
        "findings": result.get("findings", []),
        "summary": result.get("summary", ""),
        "parse_error": result.get("error") or "[codex] strict verdict unavailable",
    }


# ===========================================================================
# parse_verdict — promotion / debug consume path (ADR-106 fail-OPEN).
# ===========================================================================
#
# The non-rail promotion path (codex_invoke.py:272 -> codex.parse_verdict;
# run-promotion-gate.py future wiring) keeps the ADR-106 fail-OPEN contract:
# parse miss -> ADVISORY with a parse_error breadcrumb. Same source-of-truth
# as the strict parser (the shared object reader); the ONLY difference is that
# parse_verdict tolerates a missing/invalid verdict by coercing to ADVISORY
# WITH the object's findings/summary preserved when present, and surfaces a
# descriptive (payload-free) parse_error.
#
# CONTRACT CHANGE: parse_verdict now reads ONE structured object (the -o
# last-message content), NOT whole-stdout and NOT JSONL. Callers that previously
# passed raw whole-stdout MUST now pass the -o file content. codex_invoke.py is
# edited this ceremony to do so (it builds argv via the helper with -o <tmpfile>
# and reads that file). See open_items for the caller-migration audit.


def parse_verdict(last_message: str) -> Dict[str, Any]:
    """Parse the Codex 0.139 `-o` last-message object into a verdict envelope.

    ADR-106 fail-OPEN: parse miss coerces ``verdict`` to ``"ADVISORY"`` (Codex
    is advisory by default; never blocks on a parse miss). ``last_message`` is
    the ALREADY-REDACTED content of the ``-o/--output-last-message`` file
    (exactly one JSON object), NOT raw stdout, NOT JSONL.

    Returns a dict ALWAYS containing:
      ``verdict`` (str — coerced to ``"ADVISORY"`` on miss), ``findings``
      (List[Dict], possibly empty), ``summary`` (str), ``parse_error``
      (Optional[str]).

    NEVER raises.
    """
    result = _read_structured_verdict_object(last_message)
    if result["ok"]:
        return {
            "verdict": result["verdict"],
            "findings": result["findings"],
            "summary": result["summary"],
            "parse_error": None,
        }
    # fail-OPEN coercion per ADR-106. Preserve findings/summary if the object
    # parsed but the verdict was absent/invalid; surface a payload-free reason.
    return {
        "verdict": "ADVISORY",
        "findings": result.get("findings", []),
        "summary": result.get("summary", ""),
        "parse_error": result.get("error") or "[codex] verdict unavailable; coerced to ADVISORY",
    }


# ===========================================================================
# parse_usage_from_codex_stdout — NON-kernel promotion path ONLY (--json JSONL).
# ===========================================================================
#
# PLAN-142 §3 (R-VP-A / R-OPS-3 / R-SEC-5): the live rail DROPS usage telemetry
# (no consumer, -o only). ONLY the promotion path passes ``--json`` and wants
# token counts. On codex-cli 0.139 ``--json`` emits a JSONL EVENT STREAM on
# stdout (one JSON object per line) — so a single whole-stdout json.loads gives
# "Extra data line 2". This rewrite iterates the stream PER LINE (try/except
# each line) and takes the LAST event that carries usage/token-count fields.
#
# Field shape is defensive across event schemas: 0.139 events may nest token
# counts under a "usage" object, OR carry a token_count-style event, OR put
# input/output token integers at various keys. We accept the broadest safe set
# and coerce to non-negative ints. Absence -> all-zeros (NOT an error: usage is
# optional telemetry, must never block dispatch).


def parse_usage_from_codex_stdout(stdout: str) -> Dict[str, Any]:
    """Extract token counts + model from a codex-cli 0.139 `--json` JSONL stream.

    PLAN-142 §3 rewrite. ``stdout`` is the raw ``--json`` event stream (JSONL:
    one JSON object per line). Iterates line-by-line (try/except each line so a
    single malformed line can't abort the scan), and takes the LAST line that
    carries usage / token-count fields. Used ONLY by the non-kernel promotion
    path (codex_invoke.py / run-promotion-gate.py) — the live rail consciously
    DROPS usage (R-SEC-5).

    Returns a dict ALWAYS containing:
      ``tokens_in`` (int >=0), ``tokens_out`` (int >=0),
      ``tokens_total`` (int >=0), ``model`` (str, possibly empty),
      ``parse_error`` (Optional[str]).

    Degrades to all-zeros on absence (usage is optional telemetry; absence is
    NOT an error). NEVER raises.
    """
    fallback: Dict[str, Any] = {
        "tokens_in": 0,
        "tokens_out": 0,
        "tokens_total": 0,
        "model": "",
        "parse_error": None,
    }
    if not stdout or not isinstance(stdout, str):
        fallback["parse_error"] = "[codex] empty stdout"
        return fallback

    def _coerce_int(v: Any) -> int:
        try:
            n = int(v)
        except (TypeError, ValueError):
            return 0
        return max(0, n)

    def _usage_from_obj(obj: Dict[str, Any]) -> Optional[Dict[str, Any]]:
        """Return a usage dict from one JSONL event, or None if it carries none.

        Accepts several 0.139 event shapes defensively:
          - {"usage": {"input_tokens":..,"output_tokens":..,"total_tokens":..}}
          - {"usage": {"prompt_tokens":..,"completion_tokens":..}}  (OpenAI-ish)
          - {"type":"token_count", "input_tokens":.., "output_tokens":..}
          - flat input_tokens/output_tokens/total_tokens at top level
        Model id, if present anywhere on the event, is captured opportunistically.
        """
        if not isinstance(obj, dict):
            return None

        usage = obj.get("usage")
        src: Optional[Dict[str, Any]] = None
        if isinstance(usage, dict):
            src = usage
        elif any(
            k in obj
            for k in (
                "input_tokens",
                "output_tokens",
                "total_tokens",
                "prompt_tokens",
                "completion_tokens",
            )
        ):
            src = obj
        if src is None:
            return None

        tin = _coerce_int(src.get("input_tokens") or src.get("prompt_tokens"))
        tout = _coerce_int(src.get("output_tokens") or src.get("completion_tokens"))
        ttot = _coerce_int(src.get("total_tokens"))
        # Model id may live on the event or nested; prefer top-level "model".
        model = obj.get("model")
        if not isinstance(model, str):
            model = ""
        return {
            "tokens_in": tin,
            "tokens_out": tout,
            "tokens_total": ttot,
            "model": model,
        }

    last_usage: Optional[Dict[str, Any]] = None
    last_model: str = ""
    saw_any_json_line = False

    for raw_line in stdout.splitlines():
        line = raw_line.strip()
        if not line:
            continue
        # Strip ANSI defensively (a stray color code on a line shouldn't break
        # json.loads); fences are not expected in a --json stream but cheap.
        line = _ANSI_ESCAPE_RE.sub("", line)
        try:
            ev = json.loads(line)
        except (json.JSONDecodeError, ValueError):
            # Tolerate non-JSON lines (banners, progress) — skip, don't abort.
            continue
        if not isinstance(ev, dict):
            continue
        saw_any_json_line = True
        # Track the latest model id seen on any event.
        ev_model = ev.get("model")
        if isinstance(ev_model, str) and ev_model:
            last_model = ev_model
        u = _usage_from_obj(ev)
        if u is not None:
            last_usage = u  # keep the LAST usage-bearing event

    if last_usage is None:
        # No usage event in the stream. Not an error — return zeros, carry the
        # last-seen model id if any.
        fallback["model"] = last_model
        if not saw_any_json_line:
            fallback["parse_error"] = "[codex] no JSON lines in --json stdout"
        return fallback

    model = last_usage.get("model") or last_model or ""
    return {
        "tokens_in": last_usage["tokens_in"],
        "tokens_out": last_usage["tokens_out"],
        "tokens_total": last_usage["tokens_total"],
        "model": model,
        "parse_error": None,
    }
