# (2) REWRITE of `_invoke_codex_review` (REPLACES current L347-458).
#
# Signature CHANGE: returns the redacted CONTENT-OF-THE-`-o`-FILE as a
# str, exactly as before this returned `proc.stdout`. The consume side
# (`_decide`) therefore still receives a single str — but now it is the
# isolated last agent message (one structured JSON object) that the CLI
# wrote to the `-o` tmpfile, ALREADY single-pass redacted. The consumer
# then runs `parse_verdict_strict` on it.
#
# Why redact INSIDE this function: R-SEC-1 — the redaction must happen
# on the live-rail file-read site, BEFORE any json.loads, and the file
# is the lowest-trust surface (an untrusted binary wrote it). Doing the
# single-pass redact here keeps the trust boundary in one place and
# means `_decide` only ever sees redacted text.
# ---------------------------------------------------------------------


def _invoke_codex_review(
    tool_name: str,
    file_path: str,
    proposed_content: str,
    timeout_s: float,
) -> str:
    """Invoke Codex CLI review on codex-cli 0.139.0; return the REDACTED
    last-message file content (one structured JSON object) as a str.

    PLAN-142: the 0.139 binary REJECTS `--read-only` (exit 2). The
    verdict is obtained via `codex exec ... -o <tmpfile> --color never
    -s read-only -- "<prompt>"`, where the CLI writes ONLY the final
    agent message to <tmpfile> (verified on 0.139: a one-line JSON
    object). All flag-shape lives in the NON-kernel `_lib.codex_cli_shape`
    helper (D2); this kernel function constructs NO argv literals.

    Test-fixture path: env `CEO_PAIR_RAIL_FIXTURE_RESPONSE` short-
    circuits the subprocess invoke and returns its value verbatim
    (UNCHANGED contract — the fixture is treated as the already-isolated
    last-message string; the consume side redacts+parses it the same way
    it would a real file read, so fixture tests exercise the real parse
    path). This is the ONLY test-injection point.

    tmpfile TOCTOU hygiene (R-SEC-4 / R-VP-E):
      - `tempfile.mkdtemp(0o700)` per call (private dir we own).
      - The rail creates the file `O_CREAT|O_EXCL|0o600` at an ABSOLUTE
        path inside that dir, then closes our fd. (The CLI re-opens by
        path and overwrites — we pre-create to win the race against a
        symlink plant; we re-validate after the run anyway.)
      - After the run we REFUSE to read if the path is a symlink / not a
        regular file / not owned by us (os.lstat checks).
      - `finally:` unlink the file + rmdir the dir on EVERY exit path
        (success, read error, timeout, exception).

    Degradation matrix → CodexUnavailable / CodexMalformed (both map to
    ADVISORY in `_decide`), NEVER raise past the typed-exception wall:
      - helper unavailable               -> CodexUnavailable
      - codex binary missing             -> CodexUnavailable
      - timeout                          -> CodexTimeout
      - subprocess spawn / OS error      -> CodexUnavailable
      - exit != 0                        -> CodexUnavailable
      - file missing after exit-0        -> CodexMalformed
      - file is symlink/non-regular/
        not-owned                        -> CodexMalformed
      - file empty                       -> CodexMalformed
      - file oversize (> cap)            -> CodexMalformed

    Raises:
        CodexUnavailable, CodexTimeout, CodexMalformed (all consumed as
        ADVISORY fail-open by `_decide`).
    """
    import os as _os  # local alias to avoid any chance of shadowing
    import stat as _stat
    import tempfile as _tempfile

    fixture = _os.environ.get("CEO_PAIR_RAIL_FIXTURE_RESPONSE")
    if fixture is not None:
        # Fixture is the already-isolated last-message string. We STILL
        # single-pass redact it so the fixture path exercises the same
        # ingress trust boundary as a real file read (R-SEC-6 golden
        # fixtures land here). redact() is fail-OPEN (never raises;
        # empty in -> empty out).
        try:
            from _lib import codex_egress_redact as _redact
            return _redact.redact(fixture)
        except Exception:
            # Redactor import failure on the INGRESS path is fail-OPEN
            # to ADVISORY (NOT the ADR-114 fail-CLOSED egress contract —
            # that governs OUTGOING prompt bytes, handled below). An
            # un-redactable ingress fixture is treated as malformed so
            # the consumer degrades to ADVISORY rather than parsing
            # un-redacted text.
            raise CodexMalformed(
                "codex_egress_redact unavailable on ingress (fixture)"
            )

    codex_bin = _resolve_codex_bin()
    if codex_bin is None:
        raise CodexUnavailable("codex binary not on PATH")

    shape = _resolve_codex_cli_shape()
    if shape is None:
        # The single argv builder is gone/unimportable. Fail-OPEN to
        # ADVISORY rather than hand-roll a (possibly rejected) argv.
        raise CodexUnavailable("codex_cli_shape helper unavailable")

    prompt = _build_codex_prompt(tool_name, file_path, proposed_content)

    # -----------------------------------------------------------------
    # OUTGOING egress redaction — ADR-114, fail-CLOSED. PRESERVED
    # UNTOUCHED from the pre-PLAN-142 implementation (current L375-401).
    # This guards the PROMPT bytes that leave the framework toward the
    # Codex binary. It MUST stay fail-CLOSED: a redactor-import failure
    # REFUSES to ship the prompt.
    # -----------------------------------------------------------------
    try:
        from _lib import codex_egress_redact as _redact
        _egress_bytes = (
            len(prompt.encode("utf-8", "replace")) if isinstance(prompt, str) else 0
        )
        prompt, _egress_findings = _redact.redact_outgoing_with_findings(prompt)
    except ImportError:
        # Fail-CLOSED per ADR-114: refuse to ship unredacted prompt.
        raise CodexUnavailable(
            "codex_egress_redact unavailable — fail-CLOSED per ADR-114"
        )
    # Positive-proof emit on EVERY outbound redaction (empty allowed).
    # Fail-OPEN — wraps ONLY the emit; the redact above stays fail-CLOSED.
    try:
        from _lib import audit_emit as _ae
        _ae.emit_pair_rail_outgoing_redaction_applied(
            signal="outbound",
            match_count=len(_egress_findings),
            bytes_scanned=_egress_bytes,
            callsite="check_pair_rail.py:_invoke_codex_review",
            session_id=_os.environ.get("CLAUDE_SESSION_ID", ""),
            project=_os.environ.get("CLAUDE_PROJECT_DIR", ""),
        )
    except Exception:
        pass

    # -----------------------------------------------------------------
    # tmpfile setup — we own the dir (0o700) and pre-create the file
    # with O_EXCL|0o600 to win the symlink race; the CLI writes to it
    # by absolute path via `-o`.
    # -----------------------------------------------------------------
    tmp_dir = _tempfile.mkdtemp(prefix="ceo_pairrail_")
    try:
        _os.chmod(tmp_dir, 0o700)
    except OSError:
        # Best-effort; mkdtemp already creates 0o700 on POSIX.
        pass
    out_path = _os.path.join(tmp_dir, "codex_last_message.json")

    def _cleanup() -> None:
        """Unlink the file + rmdir the dir; never raises."""
        try:
            _os.unlink(out_path)
        except OSError:
            pass
        try:
            _os.rmdir(tmp_dir)
        except OSError:
            pass

    try:
        # Pre-create O_CREAT|O_EXCL|0o600 — if a symlink/file already
        # sits at out_path, O_EXCL fails and we treat it as unavailable.
        try:
            fd = _os.open(out_path, _os.O_CREAT | _os.O_EXCL | _os.O_WRONLY, 0o600)
            _os.close(fd)
        except FileNotFoundError:
            raise CodexUnavailable("tmpdir vanished before pre-create")
        except FileExistsError:
            raise CodexUnavailable("tmpfile pre-create lost O_EXCL race")
        except OSError as e:
            raise CodexUnavailable(
                f"tmpfile pre-create failed: {type(e).__name__}"
            )

        # Build argv via the NON-kernel helper. mode='verdict' selects the
        # `-o <out_path> --color never -s read-only` shape (NO --json on
        # the live rail — usage telemetry consciously DROPPED, R-SEC-5).
        # The helper owns flag names + model id + dead-flag migration; we
        # pass only semantic intent. (See open_items: confirm the helper's
        # exact public signature at execution.)
        try:
            cli_args = shape.build_argv(
                mode="verdict",
                prompt=prompt,
                output_last_message_path=out_path,
            )
        except Exception as e:
            raise CodexUnavailable(
                f"codex_cli_shape.build_argv failed: {type(e).__name__}"
            )
        if not isinstance(cli_args, (list, tuple)) or not cli_args:
            raise CodexUnavailable("codex_cli_shape.build_argv returned empty argv")

        cmd = [codex_bin] + list(cli_args)

        # Invoke. NO stdin prompt (the prompt is the trailing `-- "<prompt>"`
        # positional built by the helper); close stdin via input="" so the
        # child never blocks on a pipe.
        try:
            proc = subprocess.run(
                cmd,
                input="",
                capture_output=True,
                text=True,
                timeout=timeout_s,
                check=False,
            )
        except subprocess.TimeoutExpired:
            raise CodexTimeout(f"codex invoke exceeded {timeout_s}s")
        except (FileNotFoundError, PermissionError, OSError, ConnectionError) as e:
            raise CodexUnavailable(
                f"codex invoke failed: {type(e).__name__}: {e}"
            )
        except ValueError as e:
            # UnicodeDecodeError IS-A ValueError; map to malformed so
            # `_decide` degrades to ADVISORY (preserve concrete name).
            raise CodexMalformed(f"codex invoke {type(e).__name__}: {e}")

        # Post-subprocess audit breadcrumb (UNCHANGED behavior).
        try:
            from _lib import audit_emit as _ae  # type: ignore  # noqa: F811
            if hasattr(_ae, "emit_generic"):
                _ae.emit_generic(
                    "codex_invoke_dispatched", exit_code=int(proc.returncode)
                )
        except Exception:
            pass

        if proc.returncode != 0:
            # Non-zero exit → unavailable (fail-OPEN). Per the degradation
            # matrix a present-file-after-exit!=0 is IGNORED (we do not
            # read it) and cleaned up in `finally`.
            raise CodexUnavailable(
                f"codex returned exit={proc.returncode}; stderr_head="
                f"{(proc.stderr or '')[:240]!r}"
            )

        # -------------------------------------------------------------
        # Read back the `-o` file with TOCTOU re-validation. We re-stat
        # via os.lstat (NOT following symlinks) and refuse anything that
        # is not a regular file we own.
        # -------------------------------------------------------------
        try:
            st = _os.lstat(out_path)
        except FileNotFoundError:
            # File missing after exit-0 → malformed → ADVISORY.
            raise CodexMalformed("codex -o last-message file missing after exit-0")
        except OSError as e:
            raise CodexMalformed(f"codex -o lstat failed: {type(e).__name__}")

        if _stat.S_ISLNK(st.st_mode):
            raise CodexMalformed("codex -o path is a symlink; refusing to read")
        if not _stat.S_ISREG(st.st_mode):
            raise CodexMalformed("codex -o path is not a regular file; refusing")
        # Ownership: the file must be owned by us (the writer wrote at a
        # path inside our 0o700 dir, but re-check defensively).
        try:
            if st.st_uid != _os.getuid():
                raise CodexMalformed("codex -o file not owned by us; refusing")
        except AttributeError:
            # os.getuid absent (non-POSIX) — the 0o700 mkdtemp + O_EXCL
            # pre-create are the primary controls; skip the uid check.
            pass
        if st.st_size == 0:
            raise CodexMalformed("codex -o last-message file is empty")
        if st.st_size > _MAX_OUTPUT_FILE_BYTES:
            raise CodexMalformed(
                f"codex -o file oversize ({st.st_size} > {_MAX_OUTPUT_FILE_BYTES})"
            )

        # Open WITHOUT following a symlink (defense even though lstat
        # already cleared it — close the residual race). O_NOFOLLOW
        # makes the open itself refuse a symlink swapped in post-lstat.
        try:
            open_flags = _os.O_RDONLY
            if hasattr(_os, "O_NOFOLLOW"):
                open_flags |= _os.O_NOFOLLOW
            rfd = _os.open(out_path, open_flags)
        except OSError as e:
            raise CodexMalformed(f"codex -o open failed: {type(e).__name__}")
        try:
            raw_bytes = _os.read(rfd, _MAX_OUTPUT_FILE_BYTES + 1)
        finally:
            try:
                _os.close(rfd)
            except OSError:
                pass
        if len(raw_bytes) > _MAX_OUTPUT_FILE_BYTES:
            raise CodexMalformed("codex -o file grew past cap during read")

        content = raw_bytes.decode("utf-8", errors="replace")

        # -------------------------------------------------------------
        # SINGLE-PASS redact the FULL byte-string BEFORE returning (and
        # thus before any json.loads in the consumer). R-SEC-1. Ingress
        # redaction is fail-OPEN to ADVISORY (distinct from the ADR-114
        # fail-CLOSED *egress* contract above): if the redactor is
        # unavailable we refuse to hand un-redacted text to the parser
        # and degrade to ADVISORY via CodexMalformed.
        # -------------------------------------------------------------
        try:
            from _lib import codex_egress_redact as _redact  # noqa: F811
            redacted = _redact.redact(content)
        except Exception:
            raise CodexMalformed(
                "codex_egress_redact unavailable on ingress; refusing raw parse"
            )
        return redacted
    finally:
        _cleanup()

def _consume_codex_review(redacted_review: str) -> Tuple[str, Optional[str]]:
    """Derive (verdict, secondary_patch_grammar) from a REDACTED review.

    PRIMARY signal (D3): `parse_verdict_strict` — the verdict is accepted
    ONLY from a structured, schema-validated object (verdict ∈
    `_VALID_VERDICTS` = {PASS, ADVISORY, BLOCK}). A forged free-text
    'PASS' with no structured object raises `CodexJsonInvalid` →
    coerced to ADVISORY (fail-CLOSED-to-ADVISORY per R-SEC-2 / ADR-106).

    SECONDARY signal (D3): `_detect_write_shaped_patch` runs on the SAME
    redacted text as defense-in-depth ONLY — it can never *upgrade* a
    verdict to a hard block, it only annotates. The structured verdict
    is authoritative.

    Returns:
        (verdict, patch_grammar) where verdict ∈ {"PASS","ADVISORY",
        "BLOCK"} and patch_grammar is the `_detect_write_shaped_patch`
        tag (or None if clean / oversize). NEVER raises.
    """
    # Secondary defense-in-depth scan (advisory annotation only).
    try:
        patch_grammar = _detect_write_shaped_patch(redacted_review)
    except Exception:
        patch_grammar = None

    codex = _load_codex_adapter()
    if codex is None:
        # Adapter unimportable — fail-CLOSED-to-ADVISORY (we cannot
        # validate a structured verdict, so we do not trust one).
        return "ADVISORY", patch_grammar

    # `parse_verdict_strict` raises typed exceptions on any parse/schema
    # failure (CodexResponseTooLarge / CodexJsonInvalid) and carries NO
    # payload. Catch BOTH (plus a defensive catch-all) and coerce to
    # ADVISORY — the rail never blocks on an unparseable verdict.
    try:
        envelope = codex.parse_verdict_strict(redacted_review)
        verdict = envelope.get("verdict")
        if verdict not in _VALID_VERDICTS_LOCAL:
            return "ADVISORY", patch_grammar
        return str(verdict), patch_grammar
    except Exception:
        # CodexResponseTooLarge, CodexJsonInvalid, or anything else →
        # fail-CLOSED-to-ADVISORY (R-SEC-2). Importantly we do NOT fall
        # back to the free-text patch scan as the verdict — `_detect_*`
        # stays SECONDARY.
        return "ADVISORY", patch_grammar


# Mirror of `_lib.adapters.codex._VALID_VERDICTS` so the kernel can
# validate without re-importing a private name. Kept here (NOT a CLI
# literal — these are verdict labels, the trust-boundary vocabulary the
# §3 grep gate explicitly KEEPS in the kernel).
_VALID_VERDICTS_LOCAL: Tuple[str, ...] = ("PASS", "ADVISORY", "BLOCK")
